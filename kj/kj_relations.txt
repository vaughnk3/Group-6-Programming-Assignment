CREATE TABLE TVShow 
	( 
		id SMALLINT UNSIGNED NOT NULL AUTO_INCREMENT,
		name VARCHAR(30) NOT NULL,
		yearReleased TINYINT UNSIGNED,   
		description VARCHAR(300)
		PRIMARY KEY(id));
	);

CREATE TABLE Season
	(
		id SMALLINT UNSIGNED NOT NULL AUTO_INCREMENT,
		seasonName VARCHAR(30),
		seasonNumber TINYINT UNSIGNED,
		tvShowID SMALLINT UNSIGNED NOT NULL AUTO_INCREMENT,
		FOREIGN KEY(tvShowID) REFERENCES TVShow(id) ON UPDATE ON RESTRICT ON DELETE RESTRICT,
		PRIMARY KEY(id))
	);

CREATE TABLE Episode 
	(
		id SMALLINT UNSIGNED NOT NULL AUTO_INCREMENT,
		description VARCHAR(300),
		episodeNumber SMALLINT UNSIGNED,
		runtime SMALLINT UNSIGNED,
		episodeName VARCHAR(100)
		seasonID SMALLINT UNSIGNED NOT NULL AUTO_INCREMENT,
		FOREIGN KEY(seasonID) REFERENCES Season(id) ON UPDATE RESTRICT ON DELETE RESTRICT,
		PRIMARY KEY(id))
	);

CREATE TABLE Acts_In_Mov
	(
		movieId SMALLINT UNSIGNED,
		personId SMALLINT UNSIGNED
	);
