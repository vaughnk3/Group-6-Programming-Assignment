CREATE TABLE TVShow 
	( 
		id SMALLINT UNSIGNED NOT NULL AUTO_INCREMENT,
		name VARCHAR(30) NOT NULL,
		yearReleased SMALLINT UNSIGNED,   
		description VARCHAR(300),
		PRIMARY KEY(id)
	);

CREATE TABLE Season
	(
		id SMALLINT UNSIGNED NOT NULL AUTO_INCREMENT,
		seasonName VARCHAR(30),
		seasonNumber TINYINT UNSIGNED,
		tvShowID SMALLINT UNSIGNED NOT NULL,
		FOREIGN KEY(TVShowID) REFERENCES TVShow(id) ON UPDATE RESTRICT ON DELETE RESTRICT,
		PRIMARY KEY(id)
	);

CREATE TABLE Episode 
	(
		id SMALLINT UNSIGNED NOT NULL AUTO_INCREMENT,
		description VARCHAR(300),
		episodeNumber SMALLINT UNSIGNED,
		runtime SMALLINT UNSIGNED,
		episodeName VARCHAR(100),
		seasonID SMALLINT UNSIGNED NOT NULL,
		FOREIGN KEY(seasonID) REFERENCES Season(id) ON UPDATE RESTRICT ON DELETE RESTRICT,
		PRIMARY KEY(id)
	);


CREATE TABLE Person
	(
		id SMALLINT UNSIGNED NOT NULL AUTO_INCREMENT,
		firstName VARCHAR(30),
		lastName VARCHAR(30),
		sex VARCHAR(10),
		PRIMARY KEY(id)
	);

CREATE TABLE ACTS_IN_EP
	(
		EpisodeID SMALLINT UNSIGNED,
		PersonID SMALLINT UNSIGNED,
		FOREIGN KEY EpisodeID REFERENCES Episode(id) ON UPDATE RESTRICT ON DELETE RESTRICT,
		FOREIGN KEY PersonID REFERENCES Person(id) ON UPDATE RESTRICT ON DELETE RESTRICT
	);
